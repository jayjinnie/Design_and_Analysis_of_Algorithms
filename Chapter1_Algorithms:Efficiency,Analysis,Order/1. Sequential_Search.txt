void seqsearch (int n, const keytype S[], keytype x, index& location) {
  location = 1;
  while (location <= n && S[location] != x)
   location ++;
  if (location >= n)
   location = 0;
}

# Worst-case Analysis, T(n) = n
# Average-case Analysis, A(n) = (1/n) * sigma k = (1/n) * {n(n+1)}/2 = (n+1)/2
